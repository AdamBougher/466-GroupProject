Contributor(ArtistName)
    ArtistName: Primary Key

Song(SongID, Title, Genre, SongLength, ArtistName)
    SongID: Primary Key
    ArtistName: Foreign Key referencing Contributor(ArtistName)

Queue(QueueID, SongID, Username, Price)
    QueueID: Primary Key
    SongID: Foreign Key referencing Song(SongID)

PriorityQueue(QueueID, FileID, Username, Price)
    QueueID: Primary Key
    FileID: Foreign Key referencing KaraokeFiles(FileID)

KaraokeFiles(FileID, Title, SongID)
    FileID: Primary Key
    SongID: Foreign Key referencing Song(SongID)
